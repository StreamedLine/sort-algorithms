function insertionSort(arr) {
	for (var i = 1; i < arr.length; i++) {
		let currentVal = arr[i];
		let j = i - 1;
		while (j >= 0 && arr[j] > currentVal) {
			arr[j + 1] = arr[j];
			j = j - 1; 
		}
		arr[j + 1] = currentVal;
	}
	return arr
}
